1/* Well-behaved command line aliases for w64devkit
2 *
3 * Unlike batch script aliases, this program will not produce an annoying
4 * and useless "Terminate batch job (Y/N)" prompt. When compiling, define
5 * EXE as the target executable, and define CMD as the argv[0] replacement,
6 * including additional arguments. Example:
7 *
8 * $ gcc -DEXE="target.exe" -DCMD="argv0 argv1" \
9 * -Os -ffreestanding -fno-ident -fno-asynchronous-unwind-tables \
10 * -s -nostdlib -Wl,--file-alignment,16,--section-alignment,16 \
11 * -o alias.exe alias.c -lkernel32
12 *
13 * Program is compiled freestanding in order to be as small as possible.
14 *
15 * This is free and unencumbered software released into the public domain.
16 */

79mainCRTStartup(void)

81 /* Replace alias module with adjacent target. */
82 WCHAR exe[MAX_PATH + COUNTOF(LEXE)];
83 GetModuleFileNameW(0, exe, MAX_PATH);
84 WCHAR *file = findfile(exe);
85 xmemcpy(file, LEXE, sizeof(LEXE));

87 /* Produce a new command line string with new argv[0]. */
88 WCHAR *args = findargs(GetCommandLineW());
89 size_t argslen = xstrlen(args);
90 size_t cmdlen = COUNTOF(LCMD) + argslen - 1;
91 WCHAR *cmd = HeapAlloc(GetProcessHeap(), 0, sizeof(WCHAR)*cmdlen);
92 xmemcpy(cmd, LCMD, sizeof(WCHAR)*(COUNTOF(LCMD) - 1));
93 xmemcpy(cmd + COUNTOF(LCMD) - 1, args, sizeof(WCHAR)*argslen);
94
95 /* Run target with identical startup but with above changes. */
96 STARTUPINFOW si;
97 GetStartupInfoW(&si);
98 PROCESS_INFORMATION pi;
99 if (!CreateProcessW(exe, cmd, 0, 0, TRUE, 0, 0, 0, &si, &pi)) {
100 HANDLE out = GetStdHandle(STD_OUTPUT_HANDLE);
101 DWORD dummy;
102 WriteFile(out, FATAL, sizeof(FATAL), &dummy, 0);
103 return 2;
104 }
105
106 /* Wait for target to exit. */
107 DWORD ret;
108 WaitForSingleObject(pi.hProcess, INFINITE);
109 GetExitCodeProcess(pi.hProcess, &ret);
